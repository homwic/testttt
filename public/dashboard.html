Dokumentasi API Singkat

Endpoint utama:
GET /api/viu?apikey=APIKEY&username=USERNAME&domain=DOMAIN&password=PASSWORD&paket=PAKET

  • apikey : API key Anda (dari dashboard)
  • username : Username akun VIU yang ingin dibuat
  • domain : Domain email (cth: gmail.com)
  • password : Password untuk akun VIU
  • paket : Pilihan paket 1 atau 200
Contoh Code Lengkap 

// Install axios: npm install axios
const axios = require('axios');
const APIKEY = 'ISI_APIKEY_ANDA';
const USERNAME = 'ISI_USERNAME';
const DOMAIN = 'gmail.com';
const PASSWORD = 'ISI_PASSWORD';
const PAKET = '1'; // atau '200'
const url = `https://viu.lutifygame.biz.id/api/viu?apikey=${APIKEY}&username=${USERNAME}&domain=${DOMAIN}&password=${PASSWORD}&paket=${PAKET}`;
axios.get(url, { headers: { 'Content-Type': 'application/json' } })
  .then(response => { console.log(response.data); })
  .catch(error => { console.error(error.response ? error.response.data : error.message); });
            

<?php
$apikey = "ISI_APIKEY_ANDA";
$username = "ISI_USERNAME";
$domain = "gmail.com";
$password = "ISI_PASSWORD";
$paket = "1"; // atau '200'
$url = "https://viu.lutifygame.biz.id/api/viu?apikey=$apikey&username=$username&domain=$domain&password=$password&paket=$paket";
$ch = curl_init();
curl_setopt($ch, CURLOPT_URL, $url);
curl_setopt($ch, CURLOPT_RETURNTRANSFER, 1);
$response = curl_exec($ch);
if (curl_errno($ch)) {
    echo "Error: " . curl_error($ch);
} else {
    echo $response;
}
curl_close($ch);
?>
